    <script>
    function windowOpen(param){
        switch(param)
        {
            
            case 'Logout':{
                parent.window.open('http://bga.rf.gd/scripts/logout.php', '_self');
            }break;
            case 'Map':{
                parent.window.open('http://bga.rf.gd/map.php', '_self');
            }break;
            default:
                ;
        }
    }
    document.addEventListener('DOMContentLoaded', ()=>
        {
            var el = document.getElementById('body').children;
            console.log(el);
            
            for(let i=0;i<el.length;i++)
            {
                if(i == 1) continue;
                el[i].onmouseover = () => {el[i].className = 'mouseOver';}
                el[i].onmouseout = () => {el[i].className = 'mouseOut';}
            }
    
        }
    )
    </script>
